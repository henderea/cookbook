\rec{Chocolate Chip Pan Cookies}
Bake time: $15-20$ minutes @ \temp{350}
\pt{Ingredients}
\begin{itemize}
    \item 1\sfrac{1}{4} cups firmly packed brown sugar
    \item \sfrac{3}{4} stick Crisco baking sticks, butter flavored shortening
    \item 2 tablespoons milk
    \item 1 tablespoon vanilla
    \item 1 egg
    \item 1\sfrac{3}{4} cups all purpose flour
    \item 1 teaspoon salt
    \item \sfrac{3}{4} teaspoon baking soda
    \item 1 bag Semi-sweet chocolate chips
\end{itemize}
\pt{Steps}
\begin{enumerate}
    \item Heat oven to \temp{350}
    \item Use a paper towel, and scoop up some Crisco (separate from the \sfrac{3}{4} stick). Grease the baking pan (with sides).
    \item Mix brown sugar, shortening, milk, and vanilla in large mixing bowl
    \item Prepare the flour, salt, and baking soda in a measuring cup and pre-mix
    \item Beat the ingredients in the mixing bowl on medium speed with the electric mixer until light and fluffy. Beat in the egg.
    \item Mix the measuring cup ingredients into the mixing bowl, putting half in at a time and mixing it in before putting more in
    \item Using a large spoon, stir in the chocolate chips
    \item Spread the cookie dough into the pan evenly, using a wooden spatula or other tool of choice
    \item Bake 15 to 20 minutes
\end{enumerate}

\rec{Pumpkin Cookies}
Bake time: $10-12$ minutes @ \temp{350}\\
Icing Boil time: 2 minutes
\pt{Ingredients}
\spt{Cookies}
\sspt{Dry}
\begin{itemize}
    \item 2 cups sifted flower
    \item 1 teaspoon baking powder
    \item 1 teaspoon baking soda
    \item 1 teaspoon cinnamon
    \item \sfrac{1}{2} teaspoon salt
\end{itemize}
\sspt{Other}
\begin{itemize}
    \item 1 cup sugar
    \item 1 cup shortening
    \item 1 cup canned pumpkin
    \item 1 egg
    \item 1 teaspoon vanilla
\end{itemize}
\spt{Icing}
\begin{itemize}
    \item \sfrac{1}{2} cup brown sugar
    \item 3 tablespoons butter
    \item 4 tablespoons milk
    \item 1 cup powdered sugar
    \item \sfrac{1}{2} teaspoon vanilla
\end{itemize}
\clearpage
\pt{Steps}
\spt{Cookies}
\begin{enumerate}
    \item Mix dry ingredients
    \item Using small electric mixer, cream sugar and shortening, add egg and pumpkin, then blend in flour mixture
    \item Drop by tablespoons onto cookie sheet
    \item Bake $10-12$ minutes at \temp{350}
    \item Cool and frost
\end{enumerate}
\spt{Icing}
\begin{enumerate}
    \item Boil brown sugar, butter, and milk for 2 minutes
    \item Pour in powdered sugar and vanilla
    \item Whisk together until spreading consistently. It will still be somewhat thin.
    \item Spread on cookies
\end{enumerate}
